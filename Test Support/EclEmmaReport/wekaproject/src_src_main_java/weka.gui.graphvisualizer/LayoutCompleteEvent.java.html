 * @version $Revision: 1.5 $ - 24 Apr 2003 - Initial version (Ashraf M. Kibriya)
 */

  extends EventObject {

  /** for serialization */
  private static final long serialVersionUID = 6172467234026258427L;
  
  public LayoutCompleteEvent(Object source) {
    super(source);
  }
  
}
